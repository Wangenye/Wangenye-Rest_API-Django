37def ceil_div(n, d): 

38 """Return ceil(n/d), that is, the smallest integer r such that r*d >= n""" 

39 

40 if d == 0: 

41 raise ZeroDivisionError() 

42 if (n < 0) or (d < 0): 

43 raise ValueError("Non positive values") 

44 r, q = divmod(n, d) 

45 if (n != 0) and (q != 0): 

46 r += 1 

47 return r

116def GCD(x,y): 

117 """Greatest Common Denominator of :data:`x` and :data:`y`. 

118 """ 

119 

120 x = abs(x) ; y = abs(y) 

121 while x > 0: 

122 x, y = y % x, x 

123 return y

125def inverse(u, v): 

126 """The inverse of :data:`u` *mod* :data:`v`.""" 

127 

128 u3, v3 = u, v 

129 u1, v1 = 1, 0 

130 while v3 > 0: 

131 q = u3 // v3 

132 u1, v1 = v1, u1 - v1*q 

133 u3, v3 = v3, u3 - v3*q 

134 while u1<0: 

135 u1 = u1 + v 

136 return u1

155def _rabinMillerTest(n, rounds, randfunc=None): 

156 """_rabinMillerTest(n:long, rounds:int, randfunc:callable):int 

157 Tests if n is prime. 

158 Returns 0 when n is definitely composite. 

159 Returns 1 when n is probably prime. 

160 Returns 2 when n is definitely prime. 

161 

162 If randfunc is omitted, then Random.new().read is used. 

163 

164 This function is for internal use only and may be renamed or removed in 

165 the future. 

166 """ 

167 # check special cases (n==2, n even, n < 2) 

168 if n < 3 or (n & 1) == 0: 

169 return n == 2 

170 # n might be very large so it might be beneficial to precalculate n-1 

171 n_1 = n - 1 

172 # determine m and b so that 2**b * m = n - 1 and b maximal 

173 b = 0 

174 m = n_1 

175 while (m & 1) == 0: 

176 b += 1 

177 m >>= 1 

178 

179 tested = [] 

180 # we need to do at most n-2 rounds. 

181 for i in iter_range (min (rounds, n-2)): 

182 # randomly choose a < n and make sure it hasn't been tested yet 

183 a = getRandomRange (2, n, randfunc) 

184 while a in tested: 

185 a = getRandomRange (2, n, randfunc) 

186 tested.append (a) 

187 # do the rabin-miller test 

188 z = pow (a, m, n) # (a**m) % n 

189 if z == 1 or z == n_1: 

190 continue 

191 composite = 1 

192 for r in iter_range(b): 

193 z = (z * z) % n 

194 if z == 1: 

195 return 0 

196 elif z == n_1: 

197 composite = 0 

198 break 

199 if composite: 

200 return 0 

201 return 1

203def getStrongPrime(N, e=0, false_positive_prob=1e-6, randfunc=None): 

204 r""" 

205 Return a random strong *N*-bit prime number. 

206 In this context, *p* is a strong prime if *p-1* and *p+1* have at 

207 least one large prime factor. 

208 

209 Args: 

210 N (integer): the exact length of the strong prime. 

211 It must be a multiple of 128 and > 512. 

212 e (integer): if provided, the returned prime (minus 1) 

213 will be coprime to *e* and thus suitable for RSA where 

214 *e* is the public exponent. 

215 false_positive_prob (float): 

216 The statistical probability for the result not to be actually a 

217 prime. It defaults to 10\ :sup:`-6`. 

218 Note that the real probability of a false-positive is far less. This is 

219 just the mathematically provable limit. 

220 randfunc (callable): 

221 A function that takes a parameter *N* and that returns 

222 a random byte string of such length. 

223 If omitted, :func:`Cryptodome.Random.get_random_bytes` is used. 

224 Return: 

225 The new strong prime. 

226 

227 .. deprecated:: 3.0 

228 This function is for internal use only and may be renamed or removed in 

229 the future. 

230 """ 

231 

232 # This function was implemented following the 

233 # instructions found in the paper: 

234 # "FAST GENERATION OF RANDOM, STRONG RSA PRIMES" 

235 # by Robert D. Silverman 

236 # RSA Laboratories 

237 # May 17, 1997 

238 # which by the time of writing could be freely downloaded here: 

239 # http://citeseerx.ist.psu.edu/viewdoc/download?doi=10.1.1.17.2713&rep=rep1&type=pdf 

240 

241 if randfunc is None: 

242 randfunc = Random.get_random_bytes 

243 

244 # Use the accelerator if available 

245 if _fastmath is not None: 

246 return _fastmath.getStrongPrime(long(N), long(e), false_positive_prob, 

247 randfunc) 

248 

249 if (N < 512) or ((N % 128) != 0): 

250 raise ValueError ("bits must be multiple of 128 and > 512") 

251 

252 rabin_miller_rounds = int(math.ceil(-math.log(false_positive_prob)/math.log(4))) 

253 

254 # calculate range for X 

255 # lower_bound = sqrt(2) * 2^{511 + 128*x} 

256 # upper_bound = 2^{512 + 128*x} - 1 

257 x = (N - 512) >> 7; 

258 # We need to approximate the sqrt(2) in the lower_bound by an integer 

259 # expression because floating point math overflows with these numbers 

260 lower_bound = (14142135623730950489 * (2 ** (511 + 128*x))) // 10000000000000000000 

261 upper_bound = (1 << (512 + 128*x)) - 1 

262 # Randomly choose X in calculated range 

263 X = getRandomRange (lower_bound, upper_bound, randfunc) 

264 

265 # generate p1 and p2 

266 p = [0, 0] 

267 for i in (0, 1): 

268 # randomly choose 101-bit y 

269 y = getRandomNBitInteger (101, randfunc) 

270 # initialize the field for sieving 

271 field = [0] * 5 * len (sieve_base) 

272 # sieve the field 

273 for prime in sieve_base: 

274 offset = y % prime 

275 for j in iter_range((prime - offset) % prime, len (field), prime): 

276 field[j] = 1 

277 

278 # look for suitable p[i] starting at y 

279 result = 0 

280 for j in range(len(field)): 

281 composite = field[j] 

282 # look for next canidate 

283 if composite: 

284 continue 

285 tmp = y + j 

286 result = _rabinMillerTest (tmp, rabin_miller_rounds) 

287 if result > 0: 

288 p[i] = tmp 

289 break 

290 if result == 0: 

291 raise RuntimeError ("Couln't find prime in field. " 

292 "Developer: Increase field_size") 

293 

294 # Calculate R 

295 # R = (p2^{-1} mod p1) * p2 - (p1^{-1} mod p2) * p1 

296 tmp1 = inverse (p[1], p[0]) * p[1] # (p2^-1 mod p1)*p2 

297 tmp2 = inverse (p[0], p[1]) * p[0] # (p1^-1 mod p2)*p1 

298 R = tmp1 - tmp2 # (p2^-1 mod p1)*p2 - (p1^-1 mod p2)*p1 

299 

300 # search for final prime number starting by Y0 

301 # Y0 = X + (R - X mod p1p2) 

302 increment = p[0] * p[1] 

303 X = X + (R - (X % increment)) 

304 while 1: 

305 is_possible_prime = 1 

306 # first check candidate against sieve_base 

307 for prime in sieve_base: 

308 if (X % prime) == 0: 

309 is_possible_prime = 0 

310 break 

311 # if e is given make sure that e and X-1 are coprime 

312 # this is not necessarily a strong prime criterion but useful when 

313 # creating them for RSA where the p-1 and q-1 should be coprime to 

314 # the public exponent e 

315 if e and is_possible_prime: 

316 if e & 1: 

317 if GCD(e, X-1) != 1: 

318 is_possible_prime = 0 

319 else: 

320 if GCD(e, (X-1) // 2) != 1: 

321 is_possible_prime = 0 

322 

323 # do some Rabin-Miller-Tests 

324 if is_possible_prime: 

325 result = _rabinMillerTest (X, rabin_miller_rounds) 

326 if result > 0: 

327 break 

328 X += increment 

329 # abort when X has more bits than requested 

330 # TODO: maybe we shouldn't abort but rather start over. 

331 if X >= 1 << N: 

332 raise RuntimeError ("Couln't find prime in field. " 

333 "Developer: Increase field_size") 

334 return X

336def isPrime(N, false_positive_prob=1e-6, randfunc=None): 

337 r"""Test if a number *N* is a prime. 

338 

339 Args: 

340 false_positive_prob (float): 

341 The statistical probability for the result not to be actually a 

342 prime. It defaults to 10\ :sup:`-6`. 

343 Note that the real probability of a false-positive is far less. 

344 This is just the mathematically provable limit. 

345 randfunc (callable): 

346 A function that takes a parameter *N* and that returns 

347 a random byte string of such length. 

348 If omitted, :func:`Cryptodome.Random.get_random_bytes` is used. 

349 

350 Return: 

351 `True` is the input is indeed prime. 

352 """ 

353 

354 if randfunc is None: 

355 randfunc = Random.get_random_bytes 

356 

357 if _fastmath is not None: 

358 return _fastmath.isPrime(long(N), false_positive_prob, randfunc) 

359 

360 if N < 3 or N & 1 == 0: 

361 return N == 2 

362 for p in sieve_base: 

363 if N == p: 

364 return 1 

365 if N % p == 0: 

366 return 0 

367 

368 rounds = int(math.ceil(-math.log(false_positive_prob)/math.log(4))) 

369 return _rabinMillerTest(N, rounds, randfunc)

377def long_to_bytes(n, blocksize=0): 

378 """Convert an integer to a byte string. 

379 

380 In Python 3.2+, use the native method instead:: 

381 

382 >>> n.to_bytes(blocksize, 'big') 

383 

384 For instance:: 

385 

386 >>> n = 80 

387 >>> n.to_bytes(2, 'big') 

388 b'\x00P' 

389 

390 If the optional :data:`blocksize` is provided and greater than zero, 

391 the byte string is padded with binary zeros (on the front) so that 

392 the total length of the output is a multiple of blocksize. 

393 

394 If :data:`blocksize` is zero or not provided, the byte string will 

395 be of minimal length. 

396 """ 

397 # after much testing, this algorithm was deemed to be the fastest 

398 s = b'' 

399 n = int(n) 

400 pack = struct.pack 

401 while n > 0: 

402 s = pack('>I', n & 0xffffffff) + s 

403 n = n >> 32 

404 # strip off leading zeros 

405 for i in range(len(s)): 

406 if s[i] != b'\x00'[0]: 

407 break 

408 else: 

409 # only happens when n == 0 

410 s = b'\x00' 

411 i = 0 

412 s = s[i:] 

413 # add back some pad bytes. this could be done more efficiently w.r.t. the 

414 # de-padding being done above, but sigh... 

415 if blocksize > 0 and len(s) % blocksize: 

416 s = (blocksize - len(s) % blocksize) * b'\x00' + s 

417 return s

419def bytes_to_long(s): 

420 """Convert a byte string to a long integer (big endian). 

421 

422 In Python 3.2+, use the native method instead:: 

423 

424 >>> int.from_bytes(s, 'big') 

425 

426 For instance:: 

427 

428 >>> int.from_bytes(b'\x00P', 'big') 

429 80 

430 

431 This is (essentially) the inverse of :func:`long_to_bytes`. 

432 """ 

433 acc = 0 

434 

435 unpack = struct.unpack 

436 

437 # Up to Python 2.7.4, struct.unpack can't work with bytearrays nor 

438 # memoryviews 

439 if sys.version_info[0:3] < (2, 7, 4): 

440 if isinstance(s, bytearray): 

441 s = bytes(s) 

442 elif isinstance(s, memoryview): 

443 s = s.tobytes() 

444 

445 length = len(s) 

446 if length % 4: 

447 extra = (4 - length % 4) 

448 s = b'\x00' * extra + s 

449 length = length + extra 

450 for i in range(0, length, 4): 

451 acc = (acc << 32) + unpack('>I', s[i:i+4])[0] 

452 return acc
