41class PSS_SigScheme: 

42 """A signature object for ``RSASSA-PSS``. 

43 Do not instantiate directly. 

44 Use :func:`Cryptodome.Signature.pss.new`. 

45 """

47 def __init__(self, key, mgfunc, saltLen, randfunc): 

48 """Initialize this PKCS#1 PSS signature scheme object. 

49 

50 :Parameters: 

51 key : an RSA key object 

52 If a private half is given, both signature and 

53 verification are possible. 

54 If a public half is given, only verification is possible. 

55 mgfunc : callable 

56 A mask generation function that accepts two parameters: 

57 a string to use as seed, and the lenth of the mask to 

58 generate, in bytes. 

59 saltLen : integer 

60 Length of the salt, in bytes. 

61 randfunc : callable 

62 A function that returns random bytes. 

63 """ 

64 

65 self._key = key 

66 self._saltLen = saltLen 

67 self._mgfunc = mgfunc 

68 self._randfunc = randfunc 

69 

70 def can_sign(self): 

71 """Return ``True`` if this object can be used to sign messages.""" 

72 return self._key.has_private()

74 def sign(self, msg_hash): 

75 """Create the PKCS#1 PSS signature of a message. 

76 

77 This function is also called ``RSASSA-PSS-SIGN`` and 

78 it is specified in 

79 `section 8.1.1 of RFC8017 <https://tools.ietf.org/html/rfc8017#section-8.1.1>`_. 

80 

81 :parameter msg_hash: 

82 This is an object from the :mod:`Cryptodome.Hash` package. 

83 It has been used to digest the message to sign. 

84 :type msg_hash: hash object 

85 

86 :return: the signature encoded as a *byte string*. 

87 :raise ValueError: if the RSA key is not long enough for the given hash algorithm. 

88 :raise TypeError: if the RSA key has no private half. 

89 """ 

90 

91 # Set defaults for salt length and mask generation function 

92 if self._saltLen is None: 

93 sLen = msg_hash.digest_size 

94 else: 

95 sLen = self._saltLen 

96 

97 if self._mgfunc is None: 

98 mgf = lambda x, y: MGF1(x, y, msg_hash) 

99 else: 

100 mgf = self._mgfunc 

101 

102 modBits = Cryptodome.Util.number.size(self._key.n) 

103 

104 # See 8.1.1 in RFC3447 

105 k = ceil_div(modBits, 8) # k is length in bytes of the modulus 

106 # Step 1 

107 em = _EMSA_PSS_ENCODE(msg_hash, modBits-1, self._randfunc, mgf, sLen) 

108 # Step 2a (OS2IP) 

109 em_int = bytes_to_long(em) 

110 # Step 2b (RSASP1) 

111 m_int = self._key._decrypt(em_int) 

112 # Step 2c (I2OSP) 

113 signature = long_to_bytes(m_int, k) 

114 return signature

116 def verify(self, msg_hash, signature): 

117 """Check if the PKCS#1 PSS signature over a message is valid. 

118 

119 This function is also called ``RSASSA-PSS-VERIFY`` and 

120 it is specified in 

121 `section 8.1.2 of RFC8037 <https://tools.ietf.org/html/rfc8017#section-8.1.2>`_. 

122 

123 :parameter msg_hash: 

124 The hash that was carried out over the message. This is an object 

125 belonging to the :mod:`Cryptodome.Hash` module. 

126 :type parameter: hash object 

127 

128 :parameter signature: 

129 The signature that needs to be validated. 

130 :type signature: bytes 

131 

132 :raise ValueError: if the signature is not valid. 

133 """ 

134 

135 # Set defaults for salt length and mask generation function 

136 if self._saltLen is None: 

137 sLen = msg_hash.digest_size 

138 else: 

139 sLen = self._saltLen 

140 if self._mgfunc: 

141 mgf = self._mgfunc 

142 else: 

143 mgf = lambda x, y: MGF1(x, y, msg_hash) 

144 

145 modBits = Cryptodome.Util.number.size(self._key.n) 

146 

147 # See 8.1.2 in RFC3447 

148 k = ceil_div(modBits, 8) # Convert from bits to bytes 

149 # Step 1 

150 if len(signature) != k: 

151 raise ValueError("Incorrect signature") 

152 # Step 2a (O2SIP) 

153 signature_int = bytes_to_long(signature) 

154 # Step 2b (RSAVP1) 

155 em_int = self._key._encrypt(signature_int) 

156 # Step 2c (I2OSP) 

157 emLen = ceil_div(modBits - 1, 8) 

158 em = long_to_bytes(em_int, emLen) 

159 # Step 3/4 

160 _EMSA_PSS_VERIFY(msg_hash, em, modBits-1, mgf, sLen)

163def MGF1(mgfSeed, maskLen, hash_gen): 

164 """Mask Generation Function, described in `B.2.1 of RFC8017 

165 <https://tools.ietf.org/html/rfc8017>`_. 

166 

167 :param mfgSeed: 

168 seed from which the mask is generated 

169 :type mfgSeed: byte string 

170 

171 :param maskLen: 

172 intended length in bytes of the mask 

173 :type maskLen: integer 

174 

175 :param hash_gen: 

176 A module or a hash object from :mod:`Cryptodome.Hash` 

177 :type hash_object: 

178 

179 :return: the mask, as a *byte string* 

180 """ 

181 

182 T = b"" 

183 for counter in iter_range(ceil_div(maskLen, hash_gen.digest_size)): 

184 c = long_to_bytes(counter, 4) 

185 hobj = hash_gen.new() 

186 hobj.update(mgfSeed + c) 

187 T = T + hobj.digest() 

188 assert(len(T) >= maskLen) 

189 return T[:maskLen]

192def _EMSA_PSS_ENCODE(mhash, emBits, randFunc, mgf, sLen): 

193 r""" 

194 Implement the ``EMSA-PSS-ENCODE`` function, as defined 

195 in PKCS#1 v2.1 (RFC3447, 9.1.1). 

196 

197 The original ``EMSA-PSS-ENCODE`` actually accepts the message ``M`` 

198 as input, and hash it internally. Here, we expect that the message 

199 has already been hashed instead. 

200 

201 :Parameters: 

202 mhash : hash object 

203 The hash object that holds the digest of the message being signed. 

204 emBits : int 

205 Maximum length of the final encoding, in bits. 

206 randFunc : callable 

207 An RNG function that accepts as only parameter an int, and returns 

208 a string of random bytes, to be used as salt. 

209 mgf : callable 

210 A mask generation function that accepts two parameters: a string to 

211 use as seed, and the lenth of the mask to generate, in bytes. 

212 sLen : int 

213 Length of the salt, in bytes. 

214 

215 :Return: An ``emLen`` byte long string that encodes the hash 

216 (with ``emLen = \ceil(emBits/8)``). 

217 

218 :Raise ValueError: 

219 When digest or salt length are too big. 

220 """ 

221 

222 emLen = ceil_div(emBits, 8) 

223 

224 # Bitmask of digits that fill up 

225 lmask = 0 

226 for i in iter_range(8*emLen-emBits): 

227 lmask = lmask >> 1 | 0x80 

228 

229 # Step 1 and 2 have been already done 

230 # Step 3 

231 if emLen < mhash.digest_size+sLen+2: 

232 raise ValueError("Digest or salt length are too long" 

233 " for given key size.") 

234 # Step 4 

235 salt = randFunc(sLen) 

236 # Step 5 

237 m_prime = bchr(0)*8 + mhash.digest() + salt 

238 # Step 6 

239 h = mhash.new() 

240 h.update(m_prime) 

241 # Step 7 

242 ps = bchr(0)*(emLen-sLen-mhash.digest_size-2) 

243 # Step 8 

244 db = ps + bchr(1) + salt 

245 # Step 9 

246 dbMask = mgf(h.digest(), emLen-mhash.digest_size-1) 

247 # Step 10 

248 maskedDB = strxor(db, dbMask) 

249 # Step 11 

250 maskedDB = bchr(bord(maskedDB[0]) & ~lmask) + maskedDB[1:] 

251 # Step 12 

252 em = maskedDB + h.digest() + bchr(0xBC) 

253 return em

256def _EMSA_PSS_VERIFY(mhash, em, emBits, mgf, sLen): 

257 """ 

258 Implement the ``EMSA-PSS-VERIFY`` function, as defined 

259 in PKCS#1 v2.1 (RFC3447, 9.1.2). 

260 

261 ``EMSA-PSS-VERIFY`` actually accepts the message ``M`` as input, 

262 and hash it internally. Here, we expect that the message has already 

263 been hashed instead. 

264 

265 :Parameters: 

266 mhash : hash object 

267 The hash object that holds the digest of the message to be verified. 

268 em : string 

269 The signature to verify, therefore proving that the sender really 

270 signed the message that was received. 

271 emBits : int 

272 Length of the final encoding (em), in bits. 

273 mgf : callable 

274 A mask generation function that accepts two parameters: a string to 

275 use as seed, and the lenth of the mask to generate, in bytes. 

276 sLen : int 

277 Length of the salt, in bytes. 

278 

279 :Raise ValueError: 

280 When the encoding is inconsistent, or the digest or salt lengths 

281 are too big. 

282 """ 

283 

284 emLen = ceil_div(emBits, 8) 

285 

286 # Bitmask of digits that fill up 

287 lmask = 0 

288 for i in iter_range(8*emLen-emBits): 

289 lmask = lmask >> 1 | 0x80 

290 

291 # Step 1 and 2 have been already done 

292 # Step 3 

293 if emLen < mhash.digest_size+sLen+2: 

294 raise ValueError("Incorrect signature") 

295 # Step 4 

296 if ord(em[-1:]) != 0xBC: 

297 raise ValueError("Incorrect signature") 

298 # Step 5 

299 maskedDB = em[:emLen-mhash.digest_size-1] 

300 h = em[emLen-mhash.digest_size-1:-1] 

301 # Step 6 

302 if lmask & bord(em[0]): 

303 raise ValueError("Incorrect signature") 

304 # Step 7 

305 dbMask = mgf(h, emLen-mhash.digest_size-1) 

306 # Step 8 

307 db = strxor(maskedDB, dbMask) 

308 # Step 9 

309 db = bchr(bord(db[0]) & ~lmask) + db[1:] 

310 # Step 10 

311 if not db.startswith(bchr(0)*(emLen-mhash.digest_size-sLen-2) + bchr(1)): 

312 raise ValueError("Incorrect signature") 

313 # Step 11 

314 if sLen > 0: 

315 salt = db[-sLen:] 

316 else: 

317 salt = b"" 

318 # Step 12 

319 m_prime = bchr(0)*8 + mhash.digest() + salt 

320 # Step 13 

321 hobj = mhash.new() 

322 hobj.update(m_prime) 

323 hp = hobj.digest() 

324 # Step 14 

325 if h != hp: 

326 raise ValueError("Incorrect signature")

329def new(rsa_key, **kwargs): 

330 """Create an object for making or verifying PKCS#1 PSS signatures. 

331 

332 :parameter rsa_key: 

333 The RSA key to use for signing or verifying the message. 

334 This is a :class:`Cryptodome.PublicKey.RSA` object. 

335 Signing is only possible when ``rsa_key`` is a **private** RSA key. 

336 :type rsa_key: RSA object 

337 

338 :Keyword Arguments: 

339 

340 * *mask_func* (``callable``) -- 

341 A function that returns the mask (as `bytes`). 

342 It must accept two parameters: a seed (as `bytes`) 

343 and the length of the data to return. 

344 

345 If not specified, it will be the function :func:`MGF1` defined in 

346 `RFC8017 <https://tools.ietf.org/html/rfc8017#page-67>`_ and 

347 combined with the same hash algorithm applied to the 

348 message to sign or verify. 

349 

350 If you want to use a different function, for instance still :func:`MGF1` 

351 but together with another hash, you can do:: 

352 

353 from Cryptodome.Hash import SHA256 

354 from Cryptodome.Signature.pss import MGF1 

355 mgf = lambda x, y: MGF1(x, y, SHA256) 

356 

357 * *salt_bytes* (``integer``) -- 

358 Length of the salt, in bytes. 

359 It is a value between 0 and ``emLen - hLen - 2``, where ``emLen`` 

360 is the size of the RSA modulus and ``hLen`` is the size of the digest 

361 applied to the message to sign or verify. 

362 

363 The salt is generated internally, you don't need to provide it. 

364 

365 If not specified, the salt length will be ``hLen``. 

366 If it is zero, the signature scheme becomes deterministic. 

367 

368 Note that in some implementations such as OpenSSL the default 

369 salt length is ``emLen - hLen - 2`` (even though it is not more 

370 secure than ``hLen``). 

371 

372 * *rand_func* (``callable``) -- 

373 A function that returns random ``bytes``, of the desired length. 

374 The default is :func:`Cryptodome.Random.get_random_bytes`. 

375 

376 :return: a :class:`PSS_SigScheme` signature object 

377 """ 

378 

379 mask_func = kwargs.pop("mask_func", None) 

380 salt_len = kwargs.pop("salt_bytes", None) 

381 rand_func = kwargs.pop("rand_func", None) 

382 if rand_func is None: 

383 rand_func = Random.get_random_bytes 

384 if kwargs: 

385 raise ValueError("Unknown keywords: " + str(kwargs.keys())) 

386 return PSS_SigScheme(rsa_key, mask_func, salt_len, rand_func)
