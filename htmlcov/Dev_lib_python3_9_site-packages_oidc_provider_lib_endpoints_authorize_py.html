62 def _extract_params(self): 

63 """ 

64 Get all the params used by the Authorization Code Flow 

65 (and also for the Implicit and Hybrid). 

66 

67 See: http://openid.net/specs/openid-connect-core-1_0.html#AuthRequest

70 # and POST request.

72 else self.request.GET)

74 self.params['client_id'] = query_dict.get('client_id', '') 

75 self.params['redirect_uri'] = query_dict.get('redirect_uri', '') 

76 self.params['response_type'] = query_dict.get('response_type', '') 

77 self.params['scope'] = query_dict.get('scope', '').split() 

78 self.params['state'] = query_dict.get('state', '') 

79 self.params['nonce'] = query_dict.get('nonce', '')

81 self.params['prompt'] = self._allowed_prompt_params.intersection( 

82 set(query_dict.get('prompt', '').split()))

84 self.params['code_challenge'] = query_dict.get('code_challenge', '')

87 def validate_params(self): 

88 # Client validation. 

89 try: 

90 self.client = self.client_class.objects.get(client_id=self.params['client_id']) 

91 except Client.DoesNotExist: 

92 logger.debug('[Authorize] Invalid client identifier: %s', self.params['client_id']) 

93 raise ClientIdError() 

94 

95 # Redirect URI validation. 

96 if self.is_authentication and not self.params['redirect_uri']: 

97 logger.debug('[Authorize] Missing redirect uri.') 

98 raise RedirectUriError() 

99 if not (self.params['redirect_uri'] in self.client.redirect_uris): 

100 logger.debug('[Authorize] Invalid redirect uri: %s', self.params['redirect_uri']) 

101 raise RedirectUriError() 

102 

103 # Grant type validation. 

104 if not self.grant_type: 

105 logger.debug('[Authorize] Invalid response type: %s', self.params['response_type']) 

106 raise AuthorizeError( 

107 self.params['redirect_uri'], 'unsupported_response_type', self.grant_type) 

108 

109 if (not self.is_authentication and (self.grant_type == 'hybrid' or 

110 self.params['response_type'] in ['id_token', 'id_token token'])): 

111 logger.debug('[Authorize] Missing openid scope.') 

112 raise AuthorizeError(self.params['redirect_uri'], 'invalid_scope', self.grant_type) 

113 

114 # Nonce parameter validation. 

115 if self.is_authentication and self.grant_type == 'implicit' and not self.params['nonce']: 

116 raise AuthorizeError(self.params['redirect_uri'], 'invalid_request', self.grant_type) 

117 

118 # Response type parameter validation. 

119 if self.is_authentication \ 

120 and self.params['response_type'] not in self.client.response_type_values(): 

121 raise AuthorizeError(self.params['redirect_uri'], 'invalid_request', self.grant_type) 

122 

123 # PKCE validation of the transformation method. 

124 if self.params['code_challenge']: 

125 if not (self.params['code_challenge_method'] in ['plain', 'S256']): 

126 raise AuthorizeError( 

127 self.params['redirect_uri'], 'invalid_request', self.grant_type)

129 def create_response_uri(self): 

130 uri = urlsplit(self.params['redirect_uri']) 

131 query_params = parse_qs(uri.query) 

132 query_fragment = {} 

133 

134 try: 

135 if self.grant_type in ['authorization_code', 'hybrid']: 

136 code = create_code( 

137 user=self.request.user, 

138 client=self.client, 

139 scope=self.params['scope'], 

140 nonce=self.params['nonce'], 

141 is_authentication=self.is_authentication, 

142 code_challenge=self.params['code_challenge'], 

143 code_challenge_method=self.params['code_challenge_method']) 

144 code.save() 

145 

146 if self.grant_type == 'authorization_code': 

147 query_params['code'] = code.code 

148 query_params['state'] = self.params['state'] if self.params['state'] else '' 

149 elif self.grant_type in ['implicit', 'hybrid']: 

150 token = create_token( 

151 user=self.request.user, 

152 client=self.client, 

153 scope=self.params['scope']) 

154 

155 # Check if response_type must include access_token in the response. 

156 if (self.params['response_type'] in 

157 ['id_token token', 'token', 'code token', 'code id_token token']): 

158 query_fragment['access_token'] = token.access_token 

159 

160 # We don't need id_token if it's an OAuth2 request. 

161 if self.is_authentication: 

162 kwargs = { 

163 'token': token, 

164 'user': self.request.user, 

165 'aud': self.client.client_id, 

166 'nonce': self.params['nonce'], 

167 'request': self.request, 

168 'scope': self.params['scope'], 

169 } 

170 # Include at_hash when access_token is being returned. 

171 if 'access_token' in query_fragment: 

172 kwargs['at_hash'] = token.at_hash 

173 id_token_dic = create_id_token(**kwargs) 

174 

175 # Check if response_type must include id_token in the response. 

176 if self.params['response_type'] in [ 

177 'id_token', 'id_token token', 'code id_token', 'code id_token token']: 

178 query_fragment['id_token'] = encode_id_token(id_token_dic, self.client) 

179 else: 

180 id_token_dic = {} 

181 

182 # Store the token. 

183 token.id_token = id_token_dic 

184 token.save() 

185 

186 # Code parameter must be present if it's Hybrid Flow. 

187 if self.grant_type == 'hybrid': 

188 query_fragment['code'] = code.code 

189 

190 query_fragment['token_type'] = 'bearer' 

191 

192 query_fragment['expires_in'] = settings.get('OIDC_TOKEN_EXPIRE') 

193 

194 query_fragment['state'] = self.params['state'] if self.params['state'] else '' 

195 

196 if settings.get('OIDC_SESSION_MANAGEMENT_ENABLE'): 

197 # Generate client origin URI from the redirect_uri param. 

198 redirect_uri_parsed = urlsplit(self.params['redirect_uri']) 

199 client_origin = '{0}://{1}'.format( 

200 redirect_uri_parsed.scheme, redirect_uri_parsed.netloc) 

201 

202 # Create random salt. 

203 salt = md5(uuid4().hex.encode()).hexdigest() 

204 

205 # The generation of suitable Session State values is based 

206 # on a salted cryptographic hash of Client ID, origin URL, 

207 # and OP browser state. 

208 session_state = '{client_id} {origin} {browser_state} {salt}'.format( 

209 client_id=self.client.client_id, 

210 origin=client_origin, 

211 browser_state=get_browser_state_or_default(self.request), 

212 salt=salt) 

213 session_state = sha256(session_state.encode('utf-8')).hexdigest() 

214 session_state += '.' + salt 

215 if self.grant_type == 'authorization_code': 

216 query_params['session_state'] = session_state 

217 elif self.grant_type in ['implicit', 'hybrid']: 

218 query_fragment['session_state'] = session_state 

219 

220 except Exception as error: 

221 logger.exception('[Authorize] Error when trying to create response uri: %s', error) 

222 raise AuthorizeError(self.params['redirect_uri'], 'server_error', self.grant_type) 

223 

224 uri = uri._replace( 

225 query=urlencode(query_params, doseq=True), 

226 fragment=uri.fragment + urlencode(query_fragment, doseq=True))

228 return urlunsplit(uri)
